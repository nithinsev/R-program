#calculator

add <- function(x,y){
  return(x+y)
}

sub <- function(x,y){
  return(x-y)
}

mult <- function(x,y){
  return(x*y)
}

div <- function(x,y){
  return(x/y)
}
add(9,3)
sub(3,4)
mult(3,4)
div(9,3)

#kelvin_to_celsius
kelvin_to_celsius <- function(temp_in_kelvin) {
  temp_in_celsius <- temp_in_kelvin - 273.15
  return(temp_in_celsius)
}
kelvin_to_celsius(34)
